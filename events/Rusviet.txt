country_event = {
	id = SOV_civil_war.3
	title = SOV_civil_war.3.t
	desc = SOV_civil_war.3.d
	picture = GFX_report_event_hitler_handshake
	
	is_triggered_only = yes
	
	option = {
		name = SOV_civil_war.3.a
		#angers K-D and RSDLP
		set_temp_variable = {var_SOV_temp_reduce_loyalty = 5}
		SOV_reduce_rsdlp_loyalty = yes
		SOV_reduce_sr_loyalty = yes
		SOV_reduce_kd_loyalty = yes
		add_stability = -0.05
	}
	option = {
		name = SOV_civil_war.3.b
		trigger = {
			has_manpower > 14999
		}
		add_manpower = -15000
		#angers the military
		set_temp_variable = {var_SOV_temp_reduce_loyalty = 5}
		#set_temp_variable = {var_SOV_temp_reduce_controller = 5}
		set_temp_variable = {var_SOV_temp_reduce_controller = 5}
		SOV_decrease_army_control = yes
	}
	option = {
		name = SOV_civil_war.3.c
		trigger = {
			has_political_power > 149
		}
		#angers Octoberists and Tsarists
		set_temp_variable = {var_SOV_temp_reduce_loyalty = 5}
		set_temp_variable = {var_SOV_temp_increase_loyalty = 5}
		SOV_reduce_t_loyalty = yes
		SOV_increase_sr_loyalty = yes
		SOV_increase_kd_loyalty = yes
		add_political_power = -150
	}
}
