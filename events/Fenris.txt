		name = fenris.1.a
		trigger={ has_government = fenris }
		set_temp_variable = {FEN_suspicion_temp=35}

		set_global_flag = FEN_vensas_letter

		name = fenris.1.b

			NOT={
				has_government = fenris

			}
		}
	}
}
